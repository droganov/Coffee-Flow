CoffeeFlow

Simple coverflow javascript component written with coffeescript + jQuery wrapper

Setup

<-- jQuery library required -->
<script type="text/javascript" src="http://ajax.googleapis.com/ajax/libs/jquery/1.8.2/jquery.min.js"></script>

<-- Include EightMedia hammer.js if you want touchevent support -->
<script type="text/javascript" src="js/hammer.js"></script>

<-- CoffeeFlow component -->
<script type="text/javascript" src="js/coffeeflow.js"></script>

<-- CoffeeFlow jQuery wrapper (optional) -->
<script type="text/javascript" src="js/jquery.coffeeflow.js"></script>

<-- CoffeeFlow styles -->
<link rel="stylesheet" href="css/coffeeflow.css">

<-- Html -->
<div class="coffeeflow">
  <a href="img/1.jpg"><img src="img/1.jpg" alt=""></a>
  <a href="img/2.jpg"><img src="img/2.jpg" alt=""></a>
  <a href="img/3.jpg"><img src="img/3.jpg" alt=""></a>
</div>

Usage

//js
var coffeeflow = new Coffeeflow(target, options);

// jQuery
$(".coffeeflow").coffeeflow();

Options

$(".coffeeflow").coffeeflow(
  {
    borderWidth: 2, // Image border width
    debug: false, // Enables console logging 
    density: 3.2, // Use this to adjust image spacing
    defaultItem: 2, // Controls whitch items is opened on startup
    minHeight: 200, // Minimum CoffeFlow canvas height in pixels
    selectOnChange: false // If true select event is fired on change
  }
);
<-- You can also pass options to Coffeeflow with html5 -data -->
<div class="coffeeflow" data-border-width="2" data-default-item="5">
  <a href="img/1.jpg"><img src="img/1.jpg" alt=""></a>
  <a href="img/2.jpg"><img src="img/2.jpg" alt=""></a>
  <a href="img/3.jpg"><img src="img/3.jpg" alt=""></a>
</div>

Callbacks

$(".coffeeflow").coffeeflow(
  {
    blur: function (e){ console.log(e) },
    change: function (e){ console.log(e) },
    focus: function (e){ console.log(e) },
    ready: function (e){ console.log(e) },
    select: function (e){ console.log(e) }
  }
);
